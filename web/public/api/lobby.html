Lobby API Documentation

Asynchronous lobby teleportation, per-world lobby support, dynamic configuration, and comprehensive event handling for monitoring lobby-related actions

Overview

The Lobby API provides public interfaces for managing lobby teleportation and configuration. Supports both global and per-world lobby locations with asynchronous operations and animation effects.

// Basic /lobby command
public boolean onLobbyCommand(CommandSender sender, Command cmd, String label, String[] args) {
    if (!(sender instanceof Player)) {
        sender.sendMessage("Only players can use this command.");
        return true;
    }

    Player player = (Player) sender;

    // Check if lobby feature is enabled
    if (!lobbyAPI.isEnabled()) {
        player.sendMessage("¬ßcLobby system is disabled.");
        return true;
    }

    // Check if lobby exists
    lobbyAPI.hasLobby(null).thenAccept(hasLobby -> {
        if (!hasLobby) {
            player.sendMessage("¬ßcNo lobby has been set. Contact an administrator.");
            return;
        }

        // Perform teleport
        lobbyAPI.teleportToLobby(player).thenAccept(success -> {
            if (success) {
                player.sendMessage("¬ßaWhoosh! Teleported to lobby.");
            } else {
                player.sendMessage("¬ßcTeleport failed. Location may be unsafe.");
            }
        });
    });

    return true;
}

        if (distance > 100) {

Complete Usage Scenario

Multi-World Network Hub - Manages complex multi-world server networks with per-world lobbies, automated lobby returns, and usage analytics: 

package com.networkhub;

import com.serveressentials.api.lobby.*;
import com.serveressentials.api.lobby.event.*;
import org.bukkit.*;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.event.player.PlayerChangedWorldEvent;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

public class NetworkHubPlugin extends JavaPlugin implements Listener {
    private LobbyAPI lobbyAPI;
    private Map worldLobbyUsage;

    @Override
    public void onEnable() {
        this.lobbyAPI = getLobbyAPI();
        this.worldLobbyUsage = new ConcurrentHashMap<>();

        getServer().getPluginManager().registerEvents(this, this);
        getCommand("networklobby").setExecutor(this::onNetworkLobbyCommand);

        // Schedule lobby usage report
        getServer().getScheduler().runTaskTimerAsynchronously(this,
            this::sendUsageReport, 20L * 60 * 60, 20L * 60 * 60); // Hourly
    }

    @EventHandler
    public void onWorldChange(PlayerChangedWorldEvent event) {
        Player player = event.getPlayer();
        String worldName = player.getWorld().getName();

        // Auto-teleport to that world's lobby on first join
        if (!player.hasPlayedBefore()) {
            lobbyAPI.teleportToLobby(player, worldName).thenAccept(success -> {
                if (success) {
                    player.sendMessage("¬ßaWelcome to " + worldName + "! Teleported to spawn lobby.");
                }
            });
        }
    }

    @EventHandler
    public void onLobbyTeleport(LobbyTeleportEvent event) {
        Player player = event.getPlayer();
        String worldKey = event.getWorldKey();

        // Track usage
        String key = worldKey != null ? worldKey : "global";
        worldLobbyUsage.merge(key, 1, Integer::sum);

        // Custom effects for different worlds
        if ("minigames".equals(worldKey)) {
            player.playSound(player.getLocation(), Sound.ENTITY_FIREWORK_ROCKET_LAUNCH, 1.0f, 1.0f);
            player.getWorld().spawnParticle(Particle.FIREWORKS_SPARK, player.getLocation(), 50, 0.5, 0.5, 0.5, 0.1);
        }
    }

    private boolean onNetworkLobbyCommand(CommandSender sender, String[] args) {
        if (!(sender instanceof Player)) {
            sender.sendMessage("Only players can use this command.");
            return true;
        }

        Player player = (Player) sender;
        String subCmd = args[0].toLowerCase();

        switch (subCmd) {
            case "teleport":
                handleTeleport(player, args);
                break;
            case "set":
                handleSet(player, args);
                break;
            case "remove":
                handleRemove(player, args);
                break;
            case "info":
                handleInfo(player, args);
                break;
            default:
                player.sendMessage("¬ßcUnknown subcommand.");
        }

        return true;
    }

    private void handleTeleport(Player player, String[] args) {
        String worldKey = args.length > 1 ? args[1] : null;

        lobbyAPI.hasLobby(worldKey).thenAccept(hasLobby -> {
            if (!hasLobby) {
                player.sendMessage("¬ßcNo lobby found" + (worldKey != null ? " for world: " + worldKey : ""));
                return;
            }

            lobbyAPI.teleportToLobby(player, worldKey).thenAccept(success -> {
                if (success) {
                    player.sendMessage("¬ßaTeleported to lobby!");
                }
            });
        });
    }

    private void handleSet(Player player, String[] args) {
        if (!player.hasPermission("networklobby.admin.set")) {
            player.sendMessage("¬ßcNo permission.");
            return;
        }

        String worldKey = args.length > 1 ? args[1] : null;
        Location loc = player.getLocation();

        lobbyAPI.setLobby(player, loc, worldKey).thenAccept(v -> {
            player.sendMessage("¬ßaLobby set!");
            player.playSound(loc, Sound.BLOCK_BEACON_ACTIVATE, 1.0f, 1.0f);
        });
    }

    private void handleRemove(Player player, String[] args) {
        if (!player.hasPermission("networklobby.admin.remove")) {
            player.sendMessage("¬ßcNo permission.");
            return;
        }

        String worldKey = args.length > 1 ? args[1] : null;

        lobbyAPI.removeLobby(worldKey).thenAccept(v -> {
            player.sendMessage("¬ßcLobby removed!");
        });
    }

    private void handleInfo(Player player, String[] args) {
        String worldKey = args.length > 1 ? args[1] : null;

        lobbyAPI.getLobby(worldKey).thenAccept(optLoc -> {
            if (optLoc.isEmpty()) {
                player.sendMessage("¬ßcNo lobby found.");
                return;
            }

            LobbyLocation loc = optLoc.get();
            player.sendMessage("¬ß6Lobby Info:");
            player.sendMessage("¬ß7World: " + loc.getWorld());
            player.sendMessage("¬ß7Coords: " + String.format("%.1f, %.1f, %.1f", loc.getX(), loc.getY(), loc.getZ()));

            // Show usage stats
            String key = worldKey != null ? worldKey : "global";
            int usage = worldLobbyUsage.getOrDefault(key, 0);
            player.sendMessage("¬ß7Times used: " + usage);
        });
    }

    private void sendUsageReport() {
        // Send usage analytics to admins
        Bukkit.getOnlinePlayers().forEach(player -> {
            if (player.hasPermission("networklobby.stats")) {
                player.sendMessage("¬ß6¬ßlLobby Usage Report:");
                worldLobbyUsage.forEach((world, count) -> {
                    player.sendMessage("¬ß7" + world + ": " + count + " teleports");
                });
            }
        });
    }
}
Error Handling Best Practices
public void safeLobbyTeleport(Player player, String worldKey) {
    // Check if lobby system is enabled first
    if (!lobbyAPI.isEnabled()) {
        player.sendMessage("¬ßcLobby system is disabled.");
        return;
    }

    // Validate player
    if (player == null || !player.isOnline()) {
        getLogger().warning("Attempted to teleport null/offline player to lobby.");
        return;
    }

    // Check if lobby exists before attempting teleport
    lobbyAPI.hasLobby(worldKey).thenAccept(hasLobby -> {
        if (!hasLobby) {
            player.sendMessage("¬ßcNo lobby is configured" +
                (worldKey != null ? " for world: " + worldKey : "") +
                ".");
            return;
        }

        // Perform teleport with error handling
        lobbyAPI.teleportToLobby(player, worldKey)
            .exceptionally(throwable -> {
                getLogger().log(Level.SEVERE, "Lobby teleport failed for " + player.getName(), throwable);
                player.sendMessage("¬ßcTeleport failed due to internal error.");
                return false;
            })
            .thenAccept(success -> {
                if (!success) {
                    player.sendMessage("¬ßcTeleport blocked. Location may be unsafe or obstructed.");
                }
            });
    });
}
