Kit API Documentation

Programmatic control over kit claiming, cooldown management, GUI interactions, and event handling with full async support

Overview

The Kit API provides public interfaces for managing player kits, claim history, cooldown tracking, and reward distribution. All operations are asynchronous to prevent server thread blocking.

// Custom /claim command handler
public boolean onClaimCommand(CommandSender sender, Command cmd, String label, String[] args) {
    if (!(sender instanceof Player)) {
        sender.sendMessage("Only players can claim kits.");
        return true;
    }
    
    Player player = (Player) sender;
    if (args.length < 1) {
        player.sendMessage("Usage: /claim ");
        return true;
    }
    
    String kitId = args[0].toLowerCase();
    
    if (!kitAPI.isEnabled()) {
        player.sendMessage("¬ßcKit system is currently disabled.");
        return true;
    }
    
    kitAPI.claimKit(player, kitId).thenAccept(success -> {
        if (success) {
            player.sendMessage("¬ßaSuccessfully claimed kit: " + kitId);
            player.playSound(player.getLocation(), Sound.ENTITY_PLAYER_LEVELUP, 1.0f, 1.0f);
        } else {
            player.sendMessage("¬ßcFailed to claim kit. Check permissions, cooldown, or inventory space.");
        }
    }).exceptionally(throwable -> {
        getLogger().log(Level.SEVERE, "Error claiming kit for " + player.getName(), throwable);
        player.sendMessage("¬ßcAn error occurred while claiming the kit.");
        return null;
    });
    
    return true;
}

// Generate kit list for custom GUI
public void createCustomKitMenu(Player player) {
    kitAPI.getAvailableKits(player).thenAccept(kitList -> {
        if (kitList.isEmpty()) {
            player.sendMessage("¬ßcNo kits available to you.");
            return;
        }
        
        Inventory kitMenu = Bukkit.createInventory(null, 54, "¬ß6Available Kits");
        
        for (KitInfo kit : kitList) {
            // Create display item
            Material material = Material.matchMaterial(kit.getDisplayMaterial());
            if (material == null) material = Material.CHEST;
            
            ItemStack kitItem = new ItemStack(material);
            ItemMeta meta = kitItem.getItemMeta();
            meta.setDisplayName(kit.getDisplayName());
            
            // Add lore with cooldown info
            List lore = new ArrayList<>(kit.getDisplayLore());
            lore.add("");
            
            // Check if on cooldown
            kitAPI.getRemainingCooldown(player, kit.getId()).thenAccept(remainingCooldown -> {
                if (remainingCooldown > 0) {
                    String timeFormatted = formatCooldown(remainingCooldown);
                    lore.add("¬ßcCooldown: " + timeFormatted);
                    meta.addEnchant(Enchantment.LUCK, 1, true);
                    meta.addItemFlags(ItemFlag.HIDE_ENCHANTS);
                } else {
                    lore.add("¬ßaClick to claim!");
                }
                
                meta.setLore(lore);
                kitItem.setItemMeta(meta);
                
                kitMenu.setItem(kit.getSlot(), kitItem);
            });
        }
        
        player.openInventory(kitMenu);
    });
}

public boolean isKitReady(Player player, String kitId) {
    CompletableFuture canClaim = kitAPI.getRemainingCooldown(player, kitId)

Complete Usage Scenario

Advanced Kit Management Plugin - Extends Kit API with loyalty rewards, dynamic unlocking, and usage analytics: 

package com.kitmanager;

import com.serveressentials.api.kit.*;
import com.serveressentials.api.kit.event.*;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.Map;
import java.util.UUID;
import java.util.concurrent.ConcurrentHashMap;

public class AdvancedKitManager extends JavaPlugin implements Listener {
    private KitAPI kitAPI;
    private Map playerKitStreaks;
    
    @Override
    public void onEnable() {
        this.kitAPI = getKitAPI();
        this.playerKitStreaks = new ConcurrentHashMap<>();
        
        getServer().getPluginManager().registerEvents(this, this);
        getCommand("advancedkit").setExecutor(this::onAdvancedKitCommand);
        getCommand("kitstats").setExecutor(this::onKitStatsCommand);
    }

    @EventHandler
    public void onKitClaim(KitClaimEvent event) {
        Player player = event.getPlayer();
        String kitId = event.getKitId();
        UUID playerId = player.getUniqueId();
        
        int currentStreak = playerKitStreaks.getOrDefault(playerId, 0) + 1;
        playerKitStreaks.put(playerId, currentStreak);
        
        // Award loyalty bonus
        if (currentStreak % 7 == 0) {
            player.sendMessage("¬ß6¬ßlWeek Streak Bonus! ¬ß7(" + currentStreak + " days)");
            economyAPI.deposit(player, 1000.0);
            player.sendMessage("¬ßa+$1000 bonus money!");
        }
        
        // Unlock special kit after 30 days
        if (currentStreak == 30) {
            unlockSpecialKit(player, "veteran");
            player.sendMessage("¬ß6¬ßlCONGRATULATIONS! ¬ßaYou've unlocked the Veteran kit!");
        }
        
        // Log to analytics
        logKitClaim(playerId, kitId, currentStreak);
    }

    private void unlockSpecialKit(Player player, String kitId) {
        permissionAPI.addPermission(player, "essentials.kit." + kitId, true);
        player.sendMessage("¬ß7You now have access to ¬ßf/kit " + kitId);
    }

    private boolean onAdvancedKitCommand(CommandSender sender, String[] args) {
        if (!(sender instanceof Player)) return false;
        Player player = (Player) sender;
        
        if (args.length < 1) {
            player.sendMessage("Usage: /advancedkit ");
            return true;
        }
        
        String kitId = args[0];
        
        // Check if kit exists and is available
        kitAPI.getAvailableKits(player).thenAccept(kitList -> {
            KitInfo targetKit = kitList.stream()
                .filter(kit -> kit.getId().equalsIgnoreCase(kitId))
                .findFirst()
                .orElse(null);
            
            if (targetKit == null) {
                player.sendMessage("¬ßcKit not available or doesn't exist.");
                return;
            }
            
            // Check cooldown
            kitAPI.getRemainingCooldown(player, kitId).thenAccept(cooldown -> {
                if (cooldown > 0) {
                    player.sendMessage("¬ßcKit is on cooldown: " + formatTime(cooldown));
                    return;
                }
                
                // Confirm claim
                player.sendMessage("¬ß6Claiming kit: " + targetKit.getDisplayName());
                
                kitAPI.claimKit(player, kitId).thenAccept(success -> {
                    if (success) {
                        player.sendMessage("¬ßaKit claimed successfully!");
                        kitAPI.openKitPreview(player, kitId); // Show what they got
                    } else {
                        player.sendMessage("¬ßcFailed to claim kit. Check inventory space.");
                    }
                });
            });
        });
        
        return true;
    }

    private boolean onKitStatsCommand(CommandSender sender, String[] args) {
        if (!(sender instanceof Player)) return false;
        Player player = (Player) sender;
        UUID playerId = player.getUniqueId();
        
        player.sendMessage("¬ß6¬ßlYour Kit Statistics:");
        
        // Get all available kits
        kitAPI.getAvailableKits(player).thenAccept(kitList -> {
            player.sendMessage("¬ß7Total kits available: ¬ßf" + kitList.size());
            
            // Show streak
            int streak = playerKitStreaks.getOrDefault(playerId, 0);
            player.sendMessage("¬ß7Claim streak: ¬ße" + streak + " days");
            
            // Show each kit's cooldown status
            player.sendMessage("¬ß7Kit Statuses:");
            for (KitInfo kit : kitList) {
                String kitId = kit.getId();
                
                kitAPI.getRemainingCooldown(player, kitId).thenAccept(cooldown -> {
                    String status = cooldown > 0 ? "¬ßcOn CD" : "¬ßaReady";
                    String time = cooldown > 0 ? " (" + formatTime(cooldown) + ")" : "";
                    player.sendMessage("  ¬ßf- " + kit.getDisplayName() + ": " + status + time);
                });
            }
        });
        
        return true;
    }

    private String formatTime(long seconds) {
        long hours = seconds / 3600;
        long minutes = (seconds % 3600) / 60;
        long secs = seconds % 60;
        return String.format("%02d:%02d:%02d", hours, minutes, secs);
    }

    private void logKitClaim(UUID playerId, String kitId, int streak) {
        getLogger().info(String.format(
            "KIT_CLAIM: %s claimed %s (streak: %d)",
            playerId,
            kitId,
            streak
        ));
    }
}
Error Handling Best Practices
public void safeKitClaim(Player player, String kitId) {
    // Check if kit system is enabled first
    if (!kitAPI.isEnabled()) {
        player.sendMessage("¬ßcKit system is disabled.");
        return;
    }
    
    // Validate parameters
    if (player == null || !player.isOnline()) {
        getLogger().warning("Attempted to claim kit for null/offline player.");
        return;
    }
    
    if (kitId == null || kitId.isEmpty()) {
        player.sendMessage("¬ßcInvalid kit ID.");
        return;
    }
    
    // Perform claim with comprehensive error handling
    kitAPI.claimKit(player, kitId)
        .exceptionally(throwable -> {
            getLogger().log(Level.SEVERE, "Kit claim failed for " + player.getName(), throwable);
            player.sendMessage("¬ßcCritical error occurred. Contact staff.");
            return false;
        })
        .thenAccept(success -> {
            if (!success) {
                // Check why it failed
                kitAPI.getRemainingCooldown(player, kitId).thenAccept(cooldown -> {
                    if (cooldown > 0) {
                        player.sendMessage("¬ßcKit is on cooldown: " + formatCooldown(cooldown));
                    } else {
                        player.sendMessage("¬ßcFailed to claim kit. Check permissions and inventory space.");
                    }
                });
            }
        });
}

private String formatCooldown(long seconds) {
    long hours = TimeUnit.SECONDS.toHours(seconds);
    long minutes = TimeUnit.SECONDS.toMinutes(seconds) % 60;
    return String.format("%dh %dm", hours, minutes);
}
