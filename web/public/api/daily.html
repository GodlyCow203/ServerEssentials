Daily Rewards API Documentation

Complete programmatic control over daily login rewards, streak tracking, cooldown management, and custom reward integration

Overview

The Daily Rewards API provides public interfaces for managing player login rewards, claim history, cooldown tracking, and reward distribution. All operations are asynchronous to prevent server thread blocking.

Complete Usage Scenario

Streak Tracker Plugin - Tracks daily reward streaks, provides bonuses, and displays streak information: 

package com.streaktracker;

import com.serveressentials.api.daily.*;
import com.serveressentials.api.daily.event.DailyRewardClaimEvent;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.Map;
import java.util.UUID;
import java.util.concurrent.ConcurrentHashMap;

public class StreakTrackerPlugin extends JavaPlugin implements Listener {
    private DailyAPI dailyAPI;
    private Map playerStreaks;

    @Override
    public void onEnable() {
        this.dailyAPI = getDailyAPI();
        this.playerStreaks = new ConcurrentHashMap<>();
        getServer().getPluginManager().registerEvents(this, this);
        getCommand("streak").setExecutor(this::onStreakCommand);
    }

    @EventHandler
    public void onDailyClaim(DailyRewardClaimEvent event) {
        Player player = event.getPlayer();
        UUID playerId = player.getUniqueId();

        dailyAPI.getClaimedDays(playerId).thenAccept(claimedDays -> {
            int streak = calculateStreak(claimedDays);
            playerStreaks.put(playerId, streak);

            if (streak > 1) {
                double multiplier = 1.0 + (streak * 0.05);
                player.sendMessage("¬ßa¬ßlStreak Bonus! ¬ß7(" + streak + " days)");
                player.sendMessage("¬ßeYour reward value increased by " + (int)(multiplier * 100 - 100) + "%!");

                if (streak % 5 == 0) {
                    player.getInventory().addItem(new ItemStack(Material.DIAMOND, streak / 5));
                }
            }
        });
    }

    private int calculateStreak(Set claimedDays) {
        if (claimedDays.isEmpty()) return 0;
        int streak = 0;
        for (int day = 1; day <= 365; day++) {
            if (claimedDays.contains(day)) {
                streak++;
            } else {
                break;
            }
        }
        return streak;
    }

    private boolean onStreakCommand(CommandSender sender, String[] args) {
        if (!(sender instanceof Player)) return false;
        Player player = (Player) sender;
        UUID playerId = player.getUniqueId();

        dailyAPI.getClaimedDays(playerId).thenAccept(claimedDays -> {
            int streak = playerStreaks.getOrDefault(playerId, calculateStreak(claimedDays));
            player.sendMessage("¬ß6¬ßlYour Daily Streak: ¬ße" + streak + " days");

            String progressBar = createProgressBar(streak, 30);
            player.sendMessage(progressBar);

            int nextMilestone = ((streak / 7) + 1) * 7;
            player.sendMessage("¬ß7Next milestone: Day " + nextMilestone);

            dailyAPI.getTimeUntilNextClaim(playerId).thenAccept(cooldown -> {
                if (cooldown.isOnCooldown()) {
                    player.sendMessage("¬ß7Next claim in: ¬ßf" + cooldown.format());
                }
            });
        });
        return true;
    }

    private String createProgressBar(int current, int max) {
        int bars = 20;
        int filled = (current * bars) / max;
        return "¬ßa" + "|".repeat(filled) + "¬ß7" + "|".repeat(bars - filled) + " ¬ßf" + current + "/" + max;
    }
}
Error Handling Best Practices
public void safeClaimReward(Player player, int day) {
    if (!dailyAPI.isDailyEnabled()) {
        player.sendMessage("¬ßcDaily rewards are disabled on this server.");
        return;
    }

    if (day < 1 || day > 31) {
        player.sendMessage("¬ßcInvalid day number. Must be 1-31.");
        return;
    }

    dailyAPI.hasClaimedToday(player.getUniqueId()).thenAccept(hasClaimed -> {
        if (hasClaimed) {
            dailyAPI.getTimeUntilNextClaim(player.getUniqueId()).thenAccept(cooldown -> {
                player.sendMessage("¬ßcYou already claimed today!");
                player.sendMessage("¬ß7Next claim in: " + cooldown.format());
            });
            return;
        }

        dailyAPI.claimReward(player, day)
            .exceptionally(throwable -> {
                getLogger().log(Level.SEVERE, "Reward claim failed for " + player.getName(), throwable);
                player.sendMessage("¬ßcAn error occurred while claiming your reward.");
                return false;
            })
            .thenAccept(success -> {
                if (success) {
                    player.sendMessage("¬ßaReward claimed successfully!");
                    player.playSound(player.getLocation(), Sound.ENTITY_PLAYER_LEVELUP, 1.0f, 1.0f);
                } else {
                    player.sendMessage("¬ßcFailed to claim reward. Check your inventory space.");
                }
            });
    });
}
