SellGUI API Documentation

Asynchronous graphical item selling system with price management, transaction logging, and comprehensive event handling for sales analytics

Overview

The SellGUI API provides public interfaces for managing a graphical item selling system. Supports opening sell GUIs for players, checking item sellability, retrieving prices, processing bulk sell transactions, logging sales, and handling sell-related events. The system is designed to be asynchronous where appropriate to prevent blocking the main server thread during database operations.

// Calculate total value of player's inventory
public double calculateInventoryValue(Player player) {
    double totalValue = 0.0;

    for (ItemStack item : player.getInventory().getContents()) {
        if (item != null && !item.getType().isAir()) {
            double price = sellGUIAPI.getSellPrice(item.getType());
            totalValue += price * item.getAmount();
        }
    }

    return totalValue;
}

// /sellall command
public boolean onSellAllCommand(CommandSender sender) {
    if (!(sender instanceof Player)) {
        sender.sendMessage("Only players can use this command.");
        return true;
    }

    Player player = (Player) sender;

    // Collect all sellable items from inventory
    Map itemsToSell = new HashMap<>();

    for (ItemStack item : player.getInventory().getContents()) {
        if (item == null || item.getType().isAir()) continue;

        Material material = item.getType();
        if (sellGUIAPI.isSellable(material)) {
            itemsToSell.merge(material, item.getAmount(), Integer::sum);
        }
    }

    if (itemsToSell.isEmpty()) {
        player.sendMessage("¬ßcNo sellable items in your inventory.");
        return true;
    }

    // Process the sale
    player.sendMessage("¬ßeSelling all items...");

    sellGUIAPI.processSellTransaction(player, itemsToSell)
        .thenAccept(transaction -> {
            String currency = sellGUIAPI.getCurrencySymbol();
            player.sendMessage("¬ßa‚úì Sale complete! Earned: " + currency +
                String.format("%.2f", transaction.getTotalPrice()));
            player.playSound(player.getLocation(), Sound.ENTITY_EXPERIENCE_ORB_PICKUP, 1.0f, 1.0f);
        })
        .exceptionally(throwable -> {
            player.sendMessage("¬ßcSale failed: " + throwable.getMessage());
            getLogger().log(Level.WARNING, "Sell transaction failed", throwable);
            return null;
        });

    return true;
}

public class SalesAnalytics implements Listener {

Complete Usage Scenario

Advanced sell shop with tiered pricing based on player rank. 

package com.advancedsellshop;

import com.serveressentials.api.sellgui.*;
import com.serveressentials.api.sellgui.event.*;
import org.bukkit.*;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.*;

public class AdvancedSellShop extends JavaPlugin implements Listener {
    private SellGUIAPI sellGUIAPI;

    // Rank multipliers
    private Map<String, Double> rankMultipliers = Map.of(
        "default", 1.0,
        "vip", 1.25,
        "mvp", 1.5,
        "legend", 2.0
    );

    @Override
    public void onEnable() {
        this.sellGUIAPI = getSellGUIAPI(); // Obtain from plugin manager

        // Register events
        getServer().getPluginManager().registerEvents(this, this);

        // Register commands
        getCommand("sell").setExecutor(this::onSellCommand);
        getCommand("sellall").setExecutor(this::onSellAllCommand);
        getCommand("sellprices").setExecutor(this::onSellPricesCommand);
    }

    @EventHandler
    public void onItemSell(SellGUIItemSellEvent event) {
        Player player = event.getPlayer();
        String rank = getPrimaryRank(player);
        double multiplier = rankMultipliers.getOrDefault(rank, 1.0);

        if (multiplier > 1.0) {
            double bonus = (event.getTotalPrice() * multiplier) - event.getTotalPrice();
            economy.depositPlayer(player, bonus);

            player.sendMessage("¬ß6¬ßl" + rank.toUpperCase() + " BONUS! ¬ßa+" +
                sellGUIAPI.getCurrencySymbol() + String.format("%.2f", bonus));
        }
    }

    private String getPrimaryRank(Player player) {
        // Implement your permission group logic
        if (player.hasPermission("rank.legend")) return "legend";
        if (player.hasPermission("rank.mvp")) return "mvp";
        if (player.hasPermission("rank.vip")) return "vip";
        return "default";
    }
}
Error Handling Best Practices
public void safeBulkSell(Player player, Map<Material, Integer> items) {
    // Validate prerequisites
    if (!sellGUIAPI.isEnabled()) {
        player.sendMessage("¬ßcSellGUI system is disabled.");
        return;
    }

    if (player == null || !player.isOnline()) {
        getLogger().warning("Attempted sale for null/offline player.");
        return;
    }

    if (items == null || items.isEmpty()) {
        player.sendMessage("¬ßcNo items specified to sell.");
        return;
    }

    // Validate all items are sellable
    Map<Material, Integer> validItems = new HashMap<>();
    for (Map.Entry<Material, Integer> entry : items.entrySet()) {
        if (sellGUIAPI.isSellable(entry.getKey())) {
            validItems.put(entry.getKey(), entry.getValue());
        }
    }

    if (validItems.isEmpty()) {
        player.sendMessage("¬ßcNo sellable items in your selection.");
        return;
    }

    // Process with comprehensive error handling
    sellGUIAPI.processSellTransaction(player, validItems)
        .thenAccept(transaction -> {
            String currency = sellGUIAPI.getCurrencySymbol();
            player.sendMessage("¬ßa‚úì Sold " + transaction.getQuantity() + " " +
                transaction.getMaterial().name() + " for " + currency +
                String.format("%.2f", transaction.getTotalPrice()));
        })
        .exceptionally(throwable -> {
            player.sendMessage("¬ßcSale failed: " + throwable.getMessage());
            getLogger().log(Level.WARNING, "Transaction failed for " + player.getName(), throwable);
            return null;
        });
}
