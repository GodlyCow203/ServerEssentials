Back API Documentation

Complete programmatic control over /back, /lobby, and /death teleportation with async support and event handling

Overview

The Back API provides public interfaces for managing player teleportation history and saved locations. All operations are asynchronous with CompletableFuture support to prevent server thread blocking.

Complete Usage Scenario

Minigame Integration - Save player locations before arena teleport and provide custom return: 

package com.spawnrush;

import com.serveressentials.api.back.BackAPI;
import com.serveressentials.api.back.event.BackTeleportEvent;
import org.bukkit.Location;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.event.player.PlayerCommandPreprocessEvent;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.HashMap;
import java.util.Map;
import java.util.UUID;

public class SpawnRushPlugin extends JavaPlugin implements Listener {
    private BackAPI backAPI;
    private Map arenaReturnLocations;

    @Override
    public void onEnable() {
        this.backAPI = getBackAPI();
        this.arenaReturnLocations = new HashMap<>();
        getServer().getPluginManager().registerEvents(this, this);
        getCommand("leavespawnrush").setExecutor(this::onLeaveCommand);
    }

    public void joinArena(Player player, Location arenaLocation) {
        if (!backAPI.isBackEnabled()) {
            player.sendMessage("¬ßcBack system is disabled. Cannot guarantee safe return.");
            return;
        }

        Location returnLocation = player.getLocation();
        backAPI.setBackLocation(player, returnLocation).thenAccept(v -> {
            arenaReturnLocations.put(player.getUniqueId(), returnLocation.clone());
            player.teleport(arenaLocation);
            player.sendMessage("¬ßaTeleported to Spawn Rush arena!");
            player.sendMessage("¬ß7Use /back or /leavespawnrush to return.");
        }).exceptionally(throwable -> {
            player.sendMessage("¬ßcCould not save return location. Aborting.");
            return null;
        });
    }

    private boolean onLeaveCommand(CommandSender sender, Command cmd, String label, String[] args) {
        if (!(sender instanceof Player)) return false;
        Player player = (Player) sender;
        UUID playerId = player.getUniqueId();

        backAPI.hasBackLocation(player).thenAccept(hasBack -> {
            if (hasBack) {
                backAPI.teleportBack(player).thenAccept(success -> {
                    if (success) {
                        arenaReturnLocations.remove(playerId);
                    } else {
                        fallbackTeleport(player);
                    }
                });
            } else {
                fallbackTeleport(player);
            }
        });
        return true;
    }

    private void fallbackTeleport(Player player) {
        Location fallback = arenaReturnLocations.remove(player.getUniqueId());
        if (fallback != null) {
            player.teleport(fallback);
            player.sendMessage("¬ßaReturned using fallback system.");
        } else {
            player.sendMessage("¬ßcNo return location found. Contact staff.");
        }
    }

    @EventHandler
    public void onBackTeleport(BackTeleportEvent event) {
        arenaReturnLocations.remove(event.getPlayer().getUniqueId());
    }

    @EventHandler
    public void onPlayerCommand(PlayerCommandPreprocessEvent event) {
        Player player = event.getPlayer();
        if (event.getMessage().equalsIgnoreCase("/spawn") && arenaReturnLocations.containsKey(player.getUniqueId())) {
            event.setCancelled(true);
            player.sendMessage("¬ßcUse /leavespawnrush to exit the minigame first!");
        }
    }
}
Error Handling Best Practices
public void safeTeleportBack(Player player) {
    if (!backAPI.isBackEnabled()) {
        player.sendMessage("¬ßcBack teleportation is disabled on this server.");
        return;
    }

    if (player == null || !player.isOnline()) {
        getLogger().warning("Attempted to teleport null or offline player.");
        return;
    }

    backAPI.hasBackLocation(player).thenAccept(hasLocation -> {
        if (!hasLocation) {
            player.sendMessage("¬ßcYou don't have a saved back location.");
            return;
        }

        backAPI.teleportBack(player)
            .exceptionally(throwable -> {
                getLogger().log(Level.SEVERE, "Teleport failed for " + player.getName(), throwable);
                player.sendMessage("¬ßcTeleport failed due to internal error.");
                return false;
            })
            .thenAccept(success -> {
                if (success) {
                    player.sendMessage("¬ßaWhoosh!");
                } else {
                    player.sendMessage("¬ßcTeleport blocked. Location may be unsafe or obstructed.");
                }
            });
    });
}
Usage Scenario: Death Location Logger

Tracks all player death locations and provides admin commands to view last death location: 

package com.deathlogger;

import com.serveressentials.api.back.BackAPI;
import com.serveressentials.api.back.BackLocationData;
import com.serveressentials.api.back.event.BackLocationSaveEvent;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.event.entity.PlayerDeathEvent;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.Map;
import java.util.UUID;
import java.util.concurrent.ConcurrentHashMap;

public class DeathLoggerPlugin extends JavaPlugin implements Listener {
    private BackAPI backAPI;
    private Map lastDeathLocations;

    @Override
    public void onEnable() {
        this.backAPI = getBackAPI();
        this.lastDeathLocations = new ConcurrentHashMap<>();
        getServer().getPluginManager().registerEvents(this, this);
        getCommand("lastdeath").setExecutor(this::onLastDeathCommand);
    }

    @EventHandler
    public void onPlayerDeath(PlayerDeathEvent event) {
        Player player = event.getEntity();
        Location deathLoc = player.getLocation();
        BackLocationData locationData = new BackLocationData(deathLoc);
        lastDeathLocations.put(player.getUniqueId(), locationData);
        backAPI.setBackLocation(player, deathLoc);
    }

    private boolean onLastDeathCommand(CommandSender sender, Command cmd, String label, String[] args) {
        if (!sender.hasPermission("deathlogger.view")) return false;

        UUID targetId;
        if (args.length > 0) {
            targetId = Bukkit.getOfflinePlayer(args[0]).getUniqueId();
        } else if (sender instanceof Player) {
            targetId = ((Player) sender).getUniqueId();
        } else {
            sender.sendMessage("¬ßcSpecify a player.");
            return true;
        }

        BackLocationData deathLoc = lastDeathLocations.get(targetId);
        if (deathLoc != null) {
            sender.sendMessage("¬ß6Last death location:");
            sender.sendMessage("¬ßfWorld: " + deathLoc.getWorldName());
            sender.sendMessage("¬ßfCoords: " + String.format("%.1f, %.1f, %.1f",
                deathLoc.getX(), deathLoc.getY(), deathLoc.getZ()));
        } else {
            sender.sendMessage("¬ßcNo death location recorded for that player.");
        }
        return true;
    }
}
