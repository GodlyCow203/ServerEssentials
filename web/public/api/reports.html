Report API Documentation

Asynchronous player reporting system with cooldown management, validation, auto-clear capabilities, and comprehensive event handling for moderation workflows

Overview

The Report API provides public interfaces for managing player reports. Supports cooldown enforcement, duplicate prevention, staff notifications, auto-clear of stale reports, and player-specific cooldown tracking. All operations are asynchronous to prevent blocking the main server thread.

// Basic /report command handler
public boolean onReportCommand(CommandSender sender, String label, String[] args) {
    if (!(sender instanceof Player)) {
        sender.sendMessage("Only players can submit reports.");
        return true;
    }

    Player reporter = (Player) sender;

    if (args.length < 2) {
        reporter.sendMessage("Usage: /report  ");
        return true;
    }

    String targetName = args[0];
    String reason = String.join(" ", Arrays.copyOfRange(args, 1, args.length));

    if (!reportAPI.isEnabled()) {
        reporter.sendMessage("¬ßcReport system is disabled.");
        return true;
    }

    reportAPI.submitReport(reporter, targetName, reason).thenAccept(success -> {
        if (success) {
            reporter.sendMessage("¬ßaReport submitted successfully. Staff will review it soon.");

            // Notify staff
            Bukkit.getOnlinePlayers().forEach(online -> {
                if (online.hasPermission("reports.admin.notify")) {
                    online.sendMessage("¬ßc[NEW REPORT] " + targetName + " was reported by " + reporter.getName());
                    online.sendMessage("¬ß7Reason: " + reason);
                }
            });
        } else {
            reporter.sendMessage("¬ßcFailed to submit report. You may be on cooldown or target not found.");
        }
    });

    return true;
}

// Command: /clearreport 
public boolean onClearReportCommand(CommandSender sender, String[] args) {
    if (!(sender instanceof Player)) return false;

    Player staff = (Player) sender;

    if (!staff.hasPermission("reports.admin.clear")) {
        staff.sendMessage("¬ßcNo permission.");
        return true;
    }

    if (args.length < 1) {
        staff.sendMessage("Usage: /clearreport ");
        return true;
    }

    String reportId = args[0];

    reportAPI.clearReport(staff, reportId).thenAccept(v -> {
        staff.sendMessage("¬ßaReport " + reportId + " cleared.");

        // Notify reporter
        reportAPI.getReport(reportId).thenAccept(optReport -> {
            optReport.ifPresent(report -> {
                Player reporter = Bukkit.getPlayer(report.getReporterId());
                if (reporter != null && reporter.isOnline()) {
                    reporter.sendMessage("¬ß6Your report against " + report.getTargetName() + " has been reviewed and cleared.");
                }
            });
        });
    });

    return true;
}

// Check if player can report
public boolean canSubmitReport(Player player) {
    return reportAPI.getRemainingCooldown(player.getUniqueId())
        .thenApply(cooldown -> cooldown <= 0)
        .join(); // Use with caution - may block
}

// Display cooldown in report command
public void checkReportCooldown(Player player) {
    reportAPI.getRemainingCooldown(player.getUniqueId()).thenAccept(cooldown -> {
        if (cooldown > 0) {
            String timeLeft = formatCooldown(cooldown);
            player.sendMessage("¬ßcYou must wait " + timeLeft + " before submitting another report.");
        } else {
            player.sendMessage("¬ßaYou can submit a report now.");
        }
    });
}

private String formatCooldown(long seconds) {
    long minutes = seconds / 60;
    long secs = seconds % 60;
    return String.format("%02d:%02d", minutes, secs);
}

public boolean onAutoClearCommand(CommandSender sender) {

        sender.sendMessage("¬ßcNo permission.");

    Player clearer = (sender instanceof Player) ? (Player) sender : null;

    reportAPI.autoClearReports(clearer).thenAccept(count -> {
        sender.sendMessage("¬ßaAuto-cleared " + count + " old reports.");
        getLogger().info(clearerName + " auto-cleared " + count + " reports");

        if (count > 0) {

        sender.sendMessage("¬ßcAuto-clear failed: " + throwable.getMessage());
        getLogger().log(Level.SEVERE, "Report auto-clear failed", throwable);
        return null;
    });

    return true;
}

Complete Usage Scenario

Advanced moderation system with punishment integration, report prioritization, and automated investigation tools: 

package com.advancedmoderation;

import com.serveressentials.api.report.*;
import com.serveressentials.api.report.event.*;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.*;

public class AdvancedModerationPlugin extends JavaPlugin implements Listener {
    private ReportAPI reportAPI;
    private Map reportSeverity;

    @Override
    public void onEnable() {
        this.reportAPI = getReportAPI();
        this.reportSeverity = new HashMap<>();

        getServer().getPluginManager().registerEvents(this, this);
        getCommand("modreports").setExecutor(this::onModReportsCommand);
    }

    @EventHandler
    public void onReportCreate(ReportCreateEvent event) {
        ReportInfo report = event.getReportInfo();

        // Calculate severity score
        int severity = calculateSeverity(report);
        reportSeverity.put(report.getId(), severity);

        // High severity = immediate alert
        if (severity >= 10) {
            alertStaffHighPriority(report);
        }
    }

    private int calculateSeverity(ReportInfo report) {
        String lowerReason = report.getReason().toLowerCase();
        // Keywords that increase report severity
        Map<String, Integer> severityKeywords = Map.of(
            "hack", 10,
            "cheat", 10,
            "grief", 5,
            "spam", 3,
            "offensive", 7,
            "exploit", 8
        );

        int baseSeverity = 0;
        for (Map.Entry<String, Integer> entry : severityKeywords.entrySet()) {
            if (lowerReason.contains(entry.getKey())) {
                baseSeverity = Math.max(baseSeverity, entry.getValue());
            }
        }
        return baseSeverity;
    }

    private void alertStaffHighPriority(ReportInfo report) {
        String alert = "¬ß4¬ßl[URGENT] ¬ßc" + report.getTargetName() + " reported for: " + report.getReason();

        Bukkit.getOnlinePlayers().forEach(online -> {
            if (online.hasPermission("reports.urgent")) {
                online.sendMessage(alert);
                online.playSound(online.getLocation(), Sound.ENTITY_ENDER_DRAGON_GROWL, 1.0f, 1.0f);
            }
        });
    }
}
Error Handling Best Practices
public void safeReportSubmission(Player reporter, String targetName, String reason) {
    // Check if report system is enabled first
    if (!reportAPI.isEnabled()) {
        reporter.sendMessage("¬ßcReport system is disabled.");
        return;
    }

    // Validate parameters
    if (reporter == null || !reporter.isOnline()) {
        getLogger().warning("Attempted to submit report from null/offline player.");
        return;
    }

    if (targetName == null || targetName.isEmpty()) {
        reporter.sendMessage("¬ßcTarget player name cannot be empty.");
        return;
    }

    if (reason == null || reason.isEmpty()) {
        reporter.sendMessage("¬ßcReason cannot be empty.");
        return;
    }

    // Check cooldown before submission
    reportAPI.getRemainingCooldown(reporter.getUniqueId()).thenAccept(cooldown -> {
        if (cooldown > 0) {
            reporter.sendMessage("¬ßcYou're on cooldown. Wait " + formatCooldown(cooldown));
            return;
        }

        // Perform submission with error handling
        reportAPI.submitReport(reporter, targetName, reason)
            .exceptionally(throwable -> {
                getLogger().log(Level.SEVERE, "Report submission failed", throwable);
                reporter.sendMessage("¬ßcCritical error. Contact staff.");
                return false;
            })
            .thenAccept(success -> {
                if (!success) {
                    reporter.sendMessage("¬ßcFailed to submit. Target not found or duplicate report.");
                }
            });
    });
}

private String formatCooldown(long seconds) {
    if (seconds < 60) {
        return seconds + " seconds";
    }
    return String.format("%d minutes, %d seconds",
        seconds / 60, seconds % 60);
}
