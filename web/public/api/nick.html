Nick API Documentation

Asynchronous nickname changes, validation rules, cooldown management, admin controls, and comprehensive event handling for monitoring nickname activity

Overview

The Nick API provides public interfaces for managing player nicknames. Supports self-service changes, admin management, validation rules, cooldowns, and daily limits. All operations are asynchronous to prevent blocking the main server thread.

// Basic /nick command handler
public boolean onNickCommand(CommandSender sender, Command cmd, String label, String[] args) {
    if (!(sender instanceof Player)) {
        sender.sendMessage("Only players can change nicknames.");
        return true;
    }

    Player player = (Player) sender;

    if (args.length < 1) {
        player.sendMessage("Usage: /nick ");
        return true;
    }

    String newNickname = args[0];

    if (!nickAPI.isEnabled()) {
        player.sendMessage("¬ßcNickname system is disabled.");
        return true;
    }

    nickAPI.validateNickname(player, newNickname).thenAccept(isValid -> {
        if (!isValid) {
            player.sendMessage("¬ßcInvalid nickname. Check the rules with /nickrules");
            return;
        }

        nickAPI.setNickname(player, newNickname).thenAccept(success -> {
            if (success) {
                player.sendMessage("¬ßaYour nickname is now: ¬ßf" + newNickname);
                player.playSound(player.getLocation(), Sound.ENTITY_ARROW_SHOOT, 1.0f, 1.5f);
            } else {
                player.sendMessage("¬ßcFailed to set nickname. You may be on cooldown or reached daily limit.");
            }
        });
    });

    return true;
}

@EventHandler
public void onNickReset(NickResetEvent event) {
    Player initiator = event.getPlayer();
    String target = event.getTargetName();
    boolean self = event.isSelf();

    if (self) {
        getLogger().info(target + " reset their own nickname");
    } else {
        getLogger().info(initiator.getName() + " reset nickname for " + target);
    }
}

Complete Usage Scenario

Enhanced nickname management plugin with history tracking, automatic rule enforcement, and suggestion system: 

package com.nickmanager;

import com.serveressentials.api.nick.*;
import com.serveressentials.api.nick.event.*;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.*;

public class EnhancedNickPlugin extends JavaPlugin implements Listener {
    private NickAPI nickAPI;
    private Map> nicknameHistory;

    @Override
    public void onEnable() {
        this.nickAPI = getNickAPI();
        this.nicknameHistory = new HashMap<>();

        getServer().getPluginManager().registerEvents(this, this);
        loadHistory();
    }

    @EventHandler
    public void onNickSet(NickSetEvent event) {
        Player player = event.getPlayer();
        UUID playerId = player.getUniqueId();

        // Store in history
        nicknameHistory.computeIfAbsent(playerId, k -> new ArrayList<>()).add(
            String.format("%s|%s|%d",
                event.getNickInfo().getNickname(),
                event.getOldNickname(),
                System.currentTimeMillis())
        );
    }
}
Error Handling Best Practices
public void safeNicknameChange(Player player, String newNickname) {
    // Check if nickname system is enabled first
    if (!nickAPI.isEnabled()) {
        player.sendMessage("¬ßcNickname system is disabled.");
        return;
    }

    // Validate parameters
    if (player == null || !player.isOnline()) {
        getLogger().warning("Attempted to change nickname for null/offline player.");
        return;
    }

    if (newNickname == null || newNickname.isEmpty()) {
        player.sendMessage("¬ßcNickname cannot be empty.");
        return;
    }

    // Check cooldown before attempting change
    nickAPI.getRemainingCooldown(player.getUniqueId()).thenAccept(cooldown -> {
        if (cooldown > 0) {
            player.sendMessage("¬ßcYou're on cooldown. Wait " + formatCooldown(cooldown) + ".");
            return;
        }

        // Perform change with comprehensive error handling
        nickAPI.setNickname(player, newNickname)
            .exceptionally(throwable -> {
                getLogger().log(Level.SEVERE, "Nickname change failed for " + player.getName(), throwable);
                player.sendMessage("¬ßcCritical error. Contact staff.");
                return false;
            })
            .thenAccept(success -> {
                if (!success) {
                    // Check why it failed
                    nickAPI.validateNickname(player, newNickname).thenAccept(valid -> {
                        if (!valid) {
                            player.sendMessage("¬ßcNickname validation failed. Check /nickrules");
                        } else {
                            nickAPI.getDailyChanges(player.getUniqueId()).thenAccept(changes -> {
                                NickValidationRules rules = nickAPI.getValidationRules();
                                if (changes >= rules.getMaxChangesPerDay()) {
                                    player.sendMessage("¬ßcDaily limit reached. Wait until tomorrow.");
                                }
                            });
                        }
                    });
                }
            });
    });
}

private String formatCooldown(long seconds) {
    return String.format("%d seconds", seconds);
}
