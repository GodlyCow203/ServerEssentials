Shop API Documentation

Asynchronous graphical player shop system with multiple sections, pagination, custom layouts, per-item buy/sell prices, and comprehensive transaction events

Overview

The Shop API provides public interfaces for managing a comprehensive graphical player shop system with multiple sections, pagination, and custom layouts. Supports opening main shop GUIs and individual sections, per-item buy/sell prices, custom item support, and detailed transaction events. Configuration is loaded asynchronously to prevent main thread blocking.

public boolean onShopCommand(CommandSender sender, Command cmd, String label, String[] args) {
    if (!(sender instanceof Player)) {
        sender.sendMessage("Only players can use the shop.");
        return true;
    }

    Player player = (Player) sender;

    // Check if shop is enabled globally
    if (!shopAPI.isShopEnabled()) {
        player.sendMessage("¬ßcShop system is currently disabled.");

    shopAPI.openShop(player);

    getLogger().info(player.getName() + " opened the main shop.");

}

public void onSignClick(PlayerInteractEvent event) {
    if (event.getAction() != Action.RIGHT_CLICK_BLOCK) return;

    Block block = event.getClickedBlock();
    if (!(block.getState() instanceof Sign sign)) return;

    if (!sign.getLine(0).equalsIgnoreCase("[Shop]")) return;

    String sectionName = sign.getLine(1).trim();
    Player player = event.getPlayer();

    // Validate section exists before opening
    if (shopAPI.getSection(sectionName) == null) {
        player.sendMessage("¬ßcShop section '" + sectionName + "' not found.");
        return;
    }

    // Open the specific section
    shopAPI.openShopSection(player, sectionName);
    player.sendMessage("¬ßaOpening " + sectionName + " section...");
}

// /shopreload command (admin)
public boolean onShopReloadCommand(CommandSender sender) {
    if (!sender.hasPermission("shop.admin.reload")) {
        sender.sendMessage("¬ßcNo permission.");
        return true;
    }

    sender.sendMessage("¬ßeReloading shop configuration...");

    shopAPI.reloadShop().thenAccept(success -> {
        if (success) {
            sender.sendMessage("¬ßaShop configuration reloaded successfully!");

            // Display loaded sections
            Collection sections = shopAPI.getSectionNames();
            sender.sendMessage("¬ß7Loaded sections: " + String.join(", ", sections));

            // Display total items
            int totalItems = shopAPI.getSectionNames().stream()
                .mapToInt(name -> shopAPI.getSection(name).getItems().size())
                .sum();
            sender.sendMessage("¬ß7Total shop items: " + totalItems);

            getLogger().info("Shop configuration reloaded by " + sender.getName());
        } else {
            sender.sendMessage("¬ßcFailed to reload shop configuration. Check console for errors.");
        }
    }).exceptionally(throwable -> {
        sender.sendMessage("¬ßcCritical error during reload: " + throwable.getMessage());
        getLogger().log(Level.SEVERE, "Shop reload failed", throwable);
        return null;
    });

    return true;
}

// Find best buy/sell prices for a material across all sections
public void findBestPrices(Player player, Material material) {
    double bestBuyPrice = Double.MAX_VALUE;
    double bestSellPrice = 0.0;
    String bestBuySection = "";
    String bestSellSection = "";

    for (String sectionName : shopAPI.getSectionNames()) {
        ShopSection section = shopAPI.getSection(sectionName);
        if (section == null) continue;

        for (ShopItem item : section.getItems().values()) {
            if (item.getMaterial() == material) {
                // Check buy price (lowest is best)
                if (item.getBuyPrice() < bestBuyPrice) {
                    bestBuyPrice = item.getBuyPrice();
                    bestBuySection = sectionName;
                }

                // Check sell price (highest is best)
                if (item.getSellPrice() > bestSellPrice) {
                    bestSellPrice = item.getSellPrice();
                    bestSellSection = sectionName;
                }
            }
        }
    }

    String currency = "$"; // Get from ShopConfig

    player.sendMessage("¬ß6¬ßlPrice Comparison for " + material.name() + ":");
    player.sendMessage("¬ß7Best Buy: ¬ßc" + currency + bestBuyPrice + " ¬ß7(in " + bestBuySection + ")");
    player.sendMessage("¬ß7Best Sell: ¬ßa" + currency + bestSellPrice + " ¬ß7(in " + bestSellSection + ")");
}

// Create dynamic category selection GUI
public void openCategorySelector(Player player) {
    Inventory gui = Bukkit.createInventory(null, 54, "¬ß6Shop Categories");

    Collection sections = shopAPI.getSectionNames();
    int slot = 10;

    for (String sectionName : sections) {
        ShopSection section = shopAPI.getSection(sectionName);
        if (section == null) continue;

        // Create category icon
        ItemStack icon = new ItemStack(Material.CHEST);
        ItemMeta meta = icon.getItemMeta();
        meta.setDisplayName("¬ße" + sectionName.toUpperCase());

        List lore = new ArrayList<>();
        lore.add("¬ß7Items: " + section.getItems().size());
        lore.add("¬ß7Pages: " + section.getPages());
        lore.add("");
        lore.add("¬ßaClick to browse!");
        meta.setLore(lore);

        icon.setItemMeta(meta);
        gui.setItem(slot, icon);

        slot += 2; // Space out icons
        if (slot >= 44) break; // Prevent overflow
    }

    player.openInventory(gui);
}

private Map<UUID, Long> lastSaleTime = new HashMap<>();

@EventHandler
public void onShopSell(ShopSellEvent event) {
    Player player = event.getPlayer();
    UUID playerId = player.getUniqueId();

    // Prevent rapid-fire selling (potential macro)
    long currentTime = System.currentTimeMillis();
    long lastTime = lastSaleTime.getOrDefault(playerId, 0L);

    if (currentTime - lastTime < 100) { // Less than 100ms between sales
        getLogger().warning("Rapid selling detected from " + player.getName() + " - possible macro!");
    }

    lastSaleTime.put(playerId, currentTime);
}
Complete Usage Scenario

Advanced dynamic shop system with price fluctuation based on supply/demand, using all Shop API features. 

package com.dynamicshop;

import com.serveressentials.api.shop.*;
import com.serveressentials.api.shop.event.*;
import org.bukkit.*;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.inventory.ItemStack;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.*;

public class DynamicShop extends JavaPlugin implements Listener {
    private ShopAPI shopAPI;
    private Map<Material, Double> demandMultiplier = new HashMap<>();

    @Override
    public void onEnable() {
        this.shopAPI = getShopAPI(); // Obtain from plugin manager

        // Register events
        getServer().getPluginManager().registerEvents(this, this);

        // Register commands
        getCommand("market").setExecutor(this::onMarketCommand);
        getCommand("shopvalue").setExecutor(this::onShopValueCommand);

        // Load demand data
        loadDemandData();
    }

    @EventHandler
    public void onShopPurchase(ShopPurchaseEvent event) {
        // Increase demand for purchased items
        Material material = event.getItem().getType();
        double currentDemand = demandMultiplier.getOrDefault(material, 1.0);
        demandMultiplier.put(material, Math.min(currentDemand + 0.05, 3.0)); // Cap at 3x

        // Notify about price increase
        if (currentDemand > 1.5) {
            event.getPlayer().sendMessage("¬ßcDemand is high for " + material.name() + "! Prices increased.");
        }
    }
}
Error Handling Best Practices
public void safeShopOpen(Player player, String sectionName) {
    // Validate prerequisites
    if (!shopAPI.isShopEnabled()) {
        player.sendMessage("¬ßcShop system is disabled.");
        return;
    }

    if (player == null || !player.isOnline()) {
        getLogger().warning("Attempted to open shop for null/offline player.");
        return;
    }

    // Validate section if specified
    if (sectionName != null) {
        ShopSection section = shopAPI.getSection(sectionName);
        if (section == null) {
            player.sendMessage("¬ßcUnknown shop section: " + sectionName);

            // Helpful suggestion
            Collection available = shopAPI.getSectionNames();
            player.sendMessage("¬ß7Available: " + String.join(", ", available));
            return;
        }
    }

    // Open shop with error handling
    try {
        if (sectionName == null) {
            shopAPI.openShop(player);
        } else {
            shopAPI.openShopSection(player, sectionName);
        }

        getLogger().info("Opened " + (sectionName == null ? "main shop" : sectionName) +
            " for " + player.getName());

    } catch (Exception e) {
        player.sendMessage("¬ßcFailed to open shop. Please contact staff.");
        getLogger().log(Level.SEVERE, "Failed to open shop for " + player.getName(), e);
    }
}
