Economy API Documentation

Asynchronous balance operations, payment controls, transaction events, and integration support for online and offline players

Overview

The Economy API provides public interfaces for managing player economy and transactions. All financial operations are asynchronous and return CompletableFuture instances to prevent blocking the main server thread.

Complete Usage Scenario

Tax Collection Plugin - Automatically collects taxes from online players: 

package com.taxcollector;

import com.serveressentials.api.economy.EconomyAPI;
import com.serveressentials.api.economy.EconomyResponse;
import com.serveressentials.api.economy.event.EconomyWithdrawEvent;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.plugin.java.JavaPlugin;

public class TaxCollectorPlugin extends JavaPlugin implements Listener {
    private EconomyAPI economyAPI;
    private double taxRate = 0.02; // 2% tax

    @Override
    public void onEnable() {
        this.economyAPI = getEconomyAPI();

        // Tax collection every hour
        getServer().getScheduler().runTaskTimerAsynchronously(this,
            this::collectTaxes, 0L, 20L * 60 * 60);

        getServer().getPluginManager().registerEvents(this, this);
    }

    private void collectTaxes() {
        if (!economyAPI.isEnabled()) return;

        for (Player player : getServer().getOnlinePlayers()) {
            economyAPI.getBalance(player).thenAccept(balance -> {
                double taxAmount = balance * taxRate;

                if (taxAmount < 1.0) return;

                economyAPI.getPaymentSettings(player.getUniqueId().toString()).thenAccept(settings -> {
                    if (settings.isPaymentsDisabled()) {
                        getLogger().info("Skipping tax for " + player.getName() + " (payments disabled)");
                        return;
                    }

                    economyAPI.withdraw(player, taxAmount).thenAccept(response -> {
                        if (response.isSuccess()) {
                            player.sendMessage("¬ßc¬ßlTAX COLLECTED: ¬ßf-" + economyAPI.format(taxAmount));
                            depositToServer(taxAmount);
                        }
                    });
                });
            });
        }
    }

    private void depositToServer(double amount) {
        OfflinePlayer serverAccount = Bukkit.getOfflinePlayer("Server");
        economyAPI.createAccount(serverAccount).thenAccept(created -> {
            economyAPI.deposit(serverAccount, amount);
        });
    }

    @EventHandler
    public void onEconomyWithdraw(EconomyWithdrawEvent event) {
        // Log tax-related withdrawals
        if (event.getAmount() > 0 && event.getPlayer().isOnline()) {
            Player player = (Player) event.getPlayer();
            if (player.hasMetadata("tax_collector")) {
                getLogger().info("Tax collected from " + player.getName() + ": " + economyAPI.format(event.getAmount()));
            }
        }
    }
}
Error Handling Best Practices
public void safeTransaction(OfflinePlayer player, double amount, String transactionType) {
    if (!economyAPI.isEnabled()) {
        getLogger().warning("Economy is disabled. Aborting " + transactionType + ".");
        return;
    }

    if (amount <= 0) {
        getLogger().warning("Invalid transaction amount: " + amount);
        return;
    }

    if (player == null) {
        getLogger().warning("Null player in transaction: " + transactionType);
        return;
    }

    economyAPI.withdraw(player, amount)
        .exceptionally(throwable -> {
            getLogger().log(Level.SEVERE, "Critical error in " + transactionType + " for " + player.getName(), throwable);
            if (player.isOnline()) {
                ((Player) player).sendMessage("¬ßcCritical transaction error. Contact staff.");
            }
            return new EconomyResponse(amount, 0.0, false, throwable.getMessage());
        })
        .thenAccept(response -> {
            if (!response.isSuccess()) {
                getLogger().warning(
                    String.format("Transaction failed for %s: %s (Amount: %s)",
                        player.getName(),
                        response.getErrorMessage(),
                        economyAPI.format(amount)
                    )
                );
            }
        });
}
