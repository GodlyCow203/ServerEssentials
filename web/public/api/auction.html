Auction API Documentation

Programmatic control over auction listings, transactions, and economy integration for Server Essentials

Overview

The Auction API provides complete programmatic access to the auction house system. Manage listings, handle transactions, listen to events, and integrate with the server's economy.

public boolean onCommand(CommandSender sender, Command cmd, String label, String[] args) {
    if (!(sender instanceof Player)) {
        sender.sendMessage("Only players can use this command.");
        return true;
    }

    Player player = (Player) sender;
    if (auctionAPI.isAuctionEnabled()) {
        auctionAPI.openAuction(player);
    } else {
        player.sendMessage("¬ßcThe auction house is currently disabled.");

}

public void generateMarketReport(Player admin) {
    auctionAPI.getActiveItems().thenAccept(items -> {
        admin.sendMessage("=== Auction Market Report ===");
        admin.sendMessage("Total active listings: " + items.size());

        double totalValue = items.stream()
            .mapToDouble(AuctionItem::getPrice)
            .sum();

public String getTimeRemaining(AuctionItem item) {
    long now = System.currentTimeMillis();
    long diff = item.getExpiration() - now;
    if (diff <= 0) return "Expired";
    long hours = diff / (1000 * 60 * 60);
    return hours + " hours remaining";
}

Complete Usage Scenario

Dynamic Pricing Integration - Monitors auction listings to automatically adjust shop prices: 

package com.dynamicpricing;

import com.serveressentials.api.auction.AuctionAPI;
import com.serveressentials.api.auction.AuctionItem;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

public class DynamicPricingPlugin extends JavaPlugin {
    private AuctionAPI auctionAPI;
    private Map marketPrices;

    @Override
    public void onEnable() {
        this.auctionAPI = getAuctionAPI();
        this.marketPrices = new ConcurrentHashMap<>();

        // Analyze market every 5 minutes
        getServer().getScheduler().runTaskTimerAsynchronously(this,
            this::analyzeMarket, 6000L, 6000L);
    }

    private void analyzeMarket() {
        if (!auctionAPI.isAuctionEnabled()) return;

        auctionAPI.getActiveItems().thenAccept(items -> {
            marketPrices.clear();
            Map totalValue = new ConcurrentHashMap<>();
            Map itemCount = new ConcurrentHashMap<>();

            items.forEach(item -> {
                String key = item.getItem().getType().name();
                totalValue.merge(key, item.getPrice(), Double::sum);
                itemCount.merge(key, 1, Integer::sum);
            });

            itemCount.forEach((itemType, count) -> {
                double avgPrice = totalValue.get(itemType) / count;
                marketPrices.put(itemType, avgPrice);
            });
        });
    }
}
Error Handling Best Practices
public void safeAddAuctionItem(Player seller, ItemStack item, double price) {
    if (!auctionAPI.isAuctionEnabled()) {
        seller.sendMessage("¬ßcAuction house is currently disabled.");
        return;
    }

    if (price <= 0 || price > auctionAPI.getMaxPriceLimit()) {
        seller.sendMessage("¬ßcInvalid price. Must be between $1 and $" + auctionAPI.getMaxPriceLimit());
        return;
    }

    if (item == null || item.getType() == Material.AIR) {
        seller.sendMessage("¬ßcCannot auction empty item.");
        return;
    }

    auctionAPI.addAuctionItem(seller, item, price)
        .exceptionally(throwable -> {
            getLogger().log(Level.SEVERE, "Auction listing failed", throwable);
            seller.sendMessage("¬ßcAn error occurred while listing your item.");
            return false;
        })
        .thenAccept(success -> {
            if (success) {
                seller.sendMessage("¬ßaItem listed successfully!");
            } else {
                seller.sendMessage("¬ßcFailed to list item. Check inventory space and listing limits.");
            }
        });
}

Implementation Notes

All auction operations run asynchronously. Zero lag reported on servers with 100+ concurrent players.

CompletableFuture methods should be handled properly to avoid blocking the main thread.
